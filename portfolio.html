Não tenta abrir nem processar imagens — todas as respostas vêm do usuário.



Espécies consideradas:

 - Tityus serrulatus - Escorpião Amarelo

from typing import Dict, List, Tuple



def ask_yesno(prompt: str) -> bool:

    """Pergunta simples sim/não; aceita 's','sim','n','não' (ou variações em inglês)."""

    while True:

        resp = input(prompt + " (s/n): ").strip().lower()

        if resp in ("s", "sim", "y", "yes"):

            return True

        if resp in ("n", "nao", "não", "no"):

            return False

        print("Resposta inválida — digite 's' para sim ou 'n' para não.")



def score_from_answers(answers: Dict[str, bool]) -> List[Tuple[str, int, List[str]]]:

    """

    Recebe respostas booleanas e retorna lista de (espécie, pontuação, motivos).

    """

void organizarNotas(int n, float notas[]) {    // Função para ordenar as notas em ordem crescente


 
 int j, k;                                     // Variáveis de controle para o for

 float temp;                                   // Variável temporária usada para trocar valores


   
    for (j = 0; j < n; j++) {                  // Percorre todos os elementos (notas) do vetor


        
        for (k = j + 1; k < n; k++) {          // Compara cada nota com as próximas


            
            if (notas[j] > notas[k]) {  // Se a nota atual for maior que a próxima, troca as posições

                temp = notas[j];        // Guarda o valor de notas[j]

                notas[j] = notas[k];    // Coloca o valor menor em notas[j]

                notas[k] = temp;        // Coloca o valor guardado em notas[k]

            }

        }

    }

}
 


void encontrarExtremos(int n, float notas[], float *maior, float *menor) { 
 // Função para encontrar a maior e menor nota

    *maior = notas[0]; 
  // Inicializa o maior valor com a primeira nota

    *menor = notas[0]; 
  // Inicializa o menor valor com a primeira nota



    for (int i = 1; i < n; i++) {    // Percorre as notas a partir do segundo elemento
                                                  
        if (notas[i] > *maior) {     // Caso a nota atual seja maior que o maior valor encontrado

            *maior = notas[i];       // Atualiza a maior nota

        }

        if (notas[i] < *menor) {     // Caso a nota atual seja menor que o menor valor encontrado

            *menor = notas[i];       // Atualiza a menor nota

        }

    }

}
